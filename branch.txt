##在本地创建的分支
git checkout -b dev(dev是新的分支的名字，此命令代表创建一个dev分支，并且将head移到dev)
=
git branch dev
git checkout dev
#查看本地分支
git branch 
#切换分支
git checkout master
#合并分支
git merge dev (将dev分支合并到master)
#合并之后，删除分支
git branch -d dev
#解决冲突
使用git status；cat a.txt等命令，查看 
#查看分支合并情况（简介可看）
git log --graph --pretty = oneline --abbrev-commit
#分支合并图
git log --graph
#存放分支未完成工作
git stash
#查看保存的分支列表
git stash list
#恢复工作现场，并且删去stash内容
git stash pop
##远程分支
#查看远程分支
git remote 
#查看远程详细信息
git remote -v
#推送分支，将其让入远程库中,是将分支中所有文件一起推送
git push origin dev
#整理信息
git rebase
